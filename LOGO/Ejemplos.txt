CambiarColor[60,110,153];
proc cuadro(){
   for(Auxiliar=0;Auxiliar<4;Auxiliar=Auxiliar+1){
      Avanzar[100];

   }

Girar[1];

//***********************PRUEBA CON FUNCIONES******************//
func potencia () {
    retorno = 1;
    for(aux=0; aux<$2; aux=aux+1){
        retorno = retorno*$1;
    }
    return retorno;
}
for(i=0; i<24; i=i+1){
    Avanzar[potencia(i,2)];
    CambiarColor[
      5*potencia(i,1),
      5*potencia(i,2),
      5*potencia(i,3)
   ];
    Girar[90];
}

//PROCEDIMIENTOS ANIDADOS

//***************************************CARACOL***************************
proc circulo(){
   for(aux=0; aux<360; aux=aux+1){
      CambiarColor[$1*2,$2*3,$3*4];
      Avanzar[2];
      Girar[1];
   }
}
proc flor(){
   for(i=0;i<8;i=i+1){
      circulo((i+1)*30, (i+1)*32, (i+1)*60);
      Girar[45];
   }
}
flor();

//********************* TAPETE ********************************//

proc patron() {
    Girar[-90];
    for (i = 0; i < 22; i = i + 1) {
        Girar[-90];
        CambiarColor[255, 38, 190];
        Avanzar[110 - (i * 10)];
        Girar[-90];
        CambiarColor[35, 255, 75];
        Avanzar[i * 10];    
    }
}
patron();
